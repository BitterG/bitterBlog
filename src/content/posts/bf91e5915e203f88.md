# SO hook 脚本编写

## hook so文件中函数例子
```java
public native int addNumbers(int a, int b);
```

```cpp
extern "C" JNIEXPORT jint JNICALL
Java_com_example_myapplication_Java2CppTest_addNumbers(
        JNIEnv* env,
        jobject /* this */,
        jint a,
        jint b) {
    jint result = a + b;
    LOGI("addNumbers: %d + %d = %d", a, b, result);
    return result;
}
```

```cpp
__int64 __fastcall Java_com_example_myapplication_Java2CppTest_addNumbers(__int64 a1, __int64 a2, int a3, int a4)
{
  __android_log_print(4, "NativeLib", "addNumbers: %d + %d = %d", a3, a4, a4 + a3);
  return (a4 + a3);
}
```

```javascript
function hook_func_from_exports(){
    var addr = Module.findExportByName("libnative-lib.so", "Java_com_example_myapplication_Java2CppTest_addNumbers");
    console.log("addNumbers addr is:", addr);
    return addr;
}

//获取函数所在地址
var addNumbersAddr = hook_func_from_exports();
//根据地址hook
Interceptor.attach(addNumbersAddr, {
    onEnter: function (args) {
        // 参数打印
        console.log("JNIEnv *      =>", args[0]);
        console.log("jobject thiz  =>", args[1]);
        console.log("int a3        =>", args[2].toInt32());
        console.log("int a4        =>", args[3].toInt32());

        // 修改参数
        args[2] = ptr(10);  // 修改 a3
        args[3] = ptr(20);  // 修改 a4
        console.log("修改参数为: 10 + 20");
    },
    onLeave: function (retval) {
        console.log("原始返回值: " + retval.toInt32());

        // 修改返回值
        retval.replace(9999);
        console.log("修改返回值为 9999");
    }
});
```
