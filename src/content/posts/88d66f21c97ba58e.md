# hook内核函数实现文件重定向

hook目标

```c
struct file *do_filp_open(int dfd, struct filename *pathname,
		const struct open_flags *op)
```

1.引入目标内核函数所在模块的头文件

```c
#include <linux/fs.h>
//namei.c在fs目录下
```

2.定义函数指针类型

```c
//定义函数指针类型
typedef struct file *(*do_filp_open_func_t)(int dfd, struct filename *pathname,
		const struct open_flags *op);
//存储原始内核函数指针
static do_filp_open_func_t original_do_filp_open = NULL;
//备份内核函数指针
static do_filp_open_func_t backup_do_filp_open = NULL;
//替换内核函数指针
static struct file *(*replace_do_filp_open)(int dfd, struct filename *pathname,
		const struct open_flags *op);
```

3.补充缺失结构体open_flags

```c
struct open_flags {
	int open_flag;
	umode_t mode;
	int acc_mode;
	int intent;
	int lookup_flags;
};
```

4.hook与卸载hook

```c
//在init函数中
init() {
    ...
    hook_err_t err = HOOK_NO_ERR;
    //进行hook 参数: 原始指针 替换指针 备份指针
    err = hook((void *)original_do_filp_open,
            (void *)replace_do_filp_open,
            (void **)&backup_do_filp_open);
    if (err) goto out;
    ...
}
//在exit函数中
exit() {
    ...
    //卸载hook 恢复原始指针
    unhook((void *)original_do_filp_open);
    ...
}
```

void parsePath(const char *args)
{
    char buf[256];
    strncpy(buf, args, sizeof(buf) - 1);
    buf[sizeof(buf) - 1] = '\0';

    char *saveptr1;
    char *parseResult = strtok_r(buf, " ", &saveptr1); // 按空格分割每一对
    while (parseResult != NULL) {
        pr_info("parsePath token: %s\n", parseResult);

        char *saveptr2;
        char *key = strtok_r(parseResult, ":", &saveptr2);
        char *value = strtok_r(NULL, ":", &saveptr2);

        if (key && value) {
            pr_info("key = %s, value = %s\n", key, value);
            if (strcmp(key, "-p") == 0) {
                pr_info("-p = %s\n", value);
                my_atoi(value, &target_id);
                goto nextFind;
            }
            //存入数组
            strncpy((char *)orginPath[currentPathIndex], key, MAX_PATH_LEN - 1);
            orginPath[currentPathIndex][MAX_PATH_LEN - 1] = '\0';
            strncpy((char *)newPath[currentPathIndex], value, MAX_PATH_LEN - 1);
            newPath[currentPathIndex][MAX_PATH_LEN - 1] = '\0';
            currentPathIndex++;
        } else {
            pr_warn("invalid pair: %s\n", parseResult);
        }
        nextFind:
        parseResult = strtok_r(NULL, " ", &saveptr1); // 继续下一对
    }
}

int findPath(const char* currentPath) {
    pr_info("kuguaFileDirect currentPathIndex= %d", currentPathIndex);
    for (int i = 0; i < currentPathIndex; i++) {
        pr_info("kuguaFileDirect strcmp(%s, %s) result:%d \n", currentPath, orginPath[i], strcmp(currentPath, orginPath[i]));
        if (strcmp(currentPath, orginPath[i]) == 0) return i;
    }
    return -1;
}

char *strtok_r(char *s, const char *delim, char **saveptr)
{
    char *start;

    // 第一次调用
    if (s == NULL)
        s = *saveptr;

    // 跳过前导分隔符
    s += strspn(s, delim);
    if (*s == '\0') {
        *saveptr = s;
        return NULL;
    }

    // 找到下一个分隔符
    start = s;
    s = strpbrk(start, delim);
    if (s) {
        *s = '\0';       // 截断字符串
        *saveptr = s + 1; // 保存下一个开始位置
    } else {
        *saveptr = start + strlen(start); // 到字符串末尾
    }

    return start;
}

static struct file *replace_do_filp_open(int dfd, struct filename *pathname, const struct open_flags *op) {
    char buf[MAX_PATH_LEN];
    memset(buf, 0, MAX_PATH_LEN);

    const char *filename = pathname->name;
    //进程id过滤
    struct task_struct *task = current;
    pid_t pid = -1, tgid = -1;
    if (__task_pid_nr_ns) {
        pid = __task_pid_nr_ns(task, PIDTYPE_PID, 0);
        tgid = __task_pid_nr_ns(task, PIDTYPE_TGID, 0);
        pr_info("target_id: %d, current pid: %d\n", target_id, pid);
        if (target_id != -1 && pid != target_id) {
            struct file *filp = backup_do_filp_open(dfd, pathname, op);
            return filp;
        }
    }
    //查找hook
    int index = findPath(filename);
    if (index >= 0) {
        pr_info("filename : %s", filename);
        //开辟空间存储newPath
        strncpy(buf, newPath[index], sizeof(buf));
        //将新路径写入
        strncpy((char *)pathname->name, buf, strlen(buf) + 1);
    }
    
    struct file *filp = backup_do_filp_open(dfd, pathname, op);
    return filp;
}

static long syscall_hook_demo_init(const char *args, const char *event, void *__user reserved)
{
    pr_info("kpm-syscall-hook-demo init ..., args: %s\n", margs);

    __task_pid_nr_ns = (typeof(__task_pid_nr_ns))kallsyms_lookup_name("__task_pid_nr_ns");
    pr_info("kernel function __task_pid_nr_ns addr: %llx\n", __task_pid_nr_ns);
    //初始化 取内核函数指针
    original_do_filp_open = (typeof(do_filp_open_func_t))kallsyms_lookup_name("do_filp_open");
    pr_info("kernel function original_do_filp_open addr: %llx\n", original_do_filp_open);

    if (!margs) {
        pr_warn("no args specified, skip hook\n");
        return 0;
    }

    hook_err_t err = HOOK_NO_ERR;
    //进行hook 参数: 原始指针 替换指针 备份指针
    err = hook((void *)original_do_filp_open,
            (void *)replace_do_filp_open,
            (void **)&backup_do_filp_open);
    if (err) goto out;

out:
    if (err) {
        pr_err("hook original_do_filp_open error: %d\n", err);
    } else {
        pr_info("hook original_do_filp_open success\n");
    }
    return 0;
}

static long syscall_hook_demo_exit(void *__user reserved)
{
    pr_info("kpm-syscall-hook-demo exit ...\n");

    //卸载hook 恢复原始指针
    unhook((void *)original_do_filp_open);

    memset(orginPath, 0, sizeof(orginPath));
    memset(newPath, 0, sizeof(newPath));
    target_id = -1;
    return 0;

// 字符串转整数函数
// str: 输入字符串，以 '\0' 结尾
// out: 输出整数
// 返回值: 0 成功, -1 非数字, -2 溢出
int my_atoi(const char *str, int *out) {
    if (!str || !out) return -1;

    int sign = 1;
    long result = 0;

    // 跳过前导空格
    while (*str == ' ' || *str == '\t') str++;

    // 处理正负号
    if (*str == '-') {
        sign = -1;
        str++;
    } else if (*str == '+') {
        str++;
    }

    // 必须至少有一个数字
    if (*str < '0' || *str > '9') return -1;

    while (*str >= '0' && *str <= '9') {
        int digit = *str - '0';
        result = result * 10 + digit;

        // 溢出检测
        if (sign == 1 && result > 2147483647) return -2;
        if (sign == -1 && -result < (-2147483647 - 1)) return -2;

        str++;
    }

    // 尾部必须为空字符
    if (*str != '\0') return -1;

    *out = (int)(sign * result);
    return 0;
}
