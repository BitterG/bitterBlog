# Frida隐藏，防检测

## 1.端口扫描
更换frida-server启动端口即可



> 启动：frida-server -l 0.0.0.0:指定端口
>
> 转发：adb forward tcp:端口 tcp:端口
>
> 连接：frida -F/f 包名 -H 127.0.0.1:端口 -l 脚本.js
>

## 2.线程检测
针对 /proc/<pid>/task/  或  /proc/<pid>/status  查看线程，直接hook对应fgets与fopen返回伪造数据即可

```javascript
Java.perform(function () {
    // Hook libc fopen
    var fopen = Module.findExportByName("libc.so", "fopen");
    var fgets = Module.findExportByName("libc.so", "fgets");

    var fopen_impl = new NativeFunction(fopen, "pointer", ["pointer", "pointer"]);
    var fgets_impl = new NativeFunction(fgets, "pointer", ["pointer", "int", "pointer"]);

    Interceptor.replace(fopen, new NativeCallback(function (filename, mode) {
        var path = Memory.readUtf8String(filename);
        if (path.indexOf("/proc/") === 0 && path.indexOf("/task/") !== -1) {

            // 返回一个假的 FILE*
            // 用 /dev/null 来替代，避免崩溃
            var fake = fopen_impl(Memory.allocUtf8String("/dev/null"), mode);
            return fake;
        }
        return fopen_impl(filename, mode);
    }, "pointer", ["pointer", "pointer"]));

    Interceptor.replace(fgets, new NativeCallback(function (buf, size, stream) {
        var ret = fgets_impl(buf, size, stream);
        if (!ret.isNull()) {
            var line = Memory.readUtf8String(buf);
            if (line.indexOf("pool-frida") !== -1 || 
                line.indexOf("gum-js-loop") !== -1 ||
                line.indexOf("gdbus") !== -1 ||
                line.indexOf("gmain") !== -1) {
                // 直接伪造成普通线程名
                Memory.writeUtf8String(buf, "Name:\tBinder:1234\n");
            }
        }
        return ret;
    }, "pointer", ["pointer", "int", "pointer"]));
});
```

## 3.Maps记录检测
主要针对 /proc/self/maps  进行检查，对特定字符进行不存在返回的hook即可(strstr,strcmp,stat等字符对比、文件状态检查函数)

```javascript
function replace_str_maps() {
      var pt_strstr = Module.findExportByName("libc.so", 'strstr');
      var pt_strcmp = Module.findExportByName("libc.so", 'strcmp');

      Interceptor.attach(pt_strstr, {
            onEnter: function (args) {
                  var str1 = args[0].readCString();
                  var str2 = args[1].readCString();
                  if (str2.indexOf("REJECT") !== -1  ||
                  str2.indexOf("frida") !== -1 ||
                  str2.indexOf("agent") !== -1 ||
                  str2.indexOf("/data/local/tmp") !== -1){
                  this.hook = true;
                  }
            }, onLeave: function (retval) {
                  if (this.hook) {
                        retval.replace(0);
                  }
            }
      });

      Interceptor.attach(pt_strcmp, {
            onEnter: function (args) {
                  var str1 = args[0].readCString();
                  var str2 = args[1].readCString();
                  if (str2.indexOf("REJECT") !== -1  || str2.indexOf("frida") !== -1) {
                        this.hook = true;
                  }
            }, onLeave: function (retval) {
                  if (this.hook) {
                        retval.replace(0);
                  }
            }
      })

      var statPtr = Module.findExportByName("libc.so", "stat");
      Interceptor.attach(statPtr, {
      onEnter: function(args) {
            var path = args[0].readCString();
            if (path.indexOf("re.frida.server") !== -1) {
                  this.skip = true;
            }
      },
      onLeave: function(retval) {
            if (this.skip) {
                  retval.replace(-1); // 文件不存在
            }
      }
      });
}

replace_str_maps();
```

## 5.内存特征检测
同样针对 /proc/self/maps  进行关键字判断，可对fopen\strstr\ memcmp\fget\read等函数进行hook都可绕过，这里采用对fopen进行hook，也可io重定向到其他文件

```javascript
Java.perform(function() {
      Interceptor.attach(Module.findExportByName("libc.so", "fopen"), {
            onEnter: function(args) {
                  this.path = args[0].readCString();
            },
            onLeave: function(retval) {
                  if (this.path.indexOf("/proc/self/maps") >= 0) {
                        retval.replace(ptr(0)); // 返回 NULL，表示打开失败
                  }
            }
      });
})
```

## 6.trace状态检测
 针对/proc/<pid>/status  或  /proc/<pid>/task/<tid>/status  的 TracerPid  字段判断

> Name:   myapp
>
> State:  R (running)
>
> Tgid:   1234
>
> Pid:    1234
>
> PPid:   1233
>
> TracerPid: 0  <-获取时用过hook返回为0即可
>
> ...简化
>

```javascript
Java.perform(function() {
    var fopen = Module.findExportByName("libc.so", "fopen");
    var fgets = Module.findExportByName("libc.so", "fgets");

    var fopen_impl = new NativeFunction(fopen, "pointer", ["pointer", "pointer"]);
    var fgets_impl = new NativeFunction(fgets, "pointer", ["pointer", "int", "pointer"]);

    // hook fopen
    Interceptor.replace(fopen, new NativeCallback(function(filename, mode) {
        var path = Memory.readUtf8String(filename);
        if (path.indexOf("/proc/") === 0 && path.indexOf("/status") !== -1) {
            // 返回真实文件即可
            return fopen_impl(filename, mode);
        }
        return fopen_impl(filename, mode);
    }, "pointer", ["pointer", "pointer"]));

    // hook fgets
    Interceptor.replace(fgets, new NativeCallback(function(buf, size, stream) {
        var ret = fgets_impl(buf, size, stream);
        if (!ret.isNull()) {
            var line = Memory.readUtf8String(buf);
            if (line.indexOf("TracerPid:") !== -1) {
                // 伪造 TracerPid 为 0
                Memory.writeUtf8String(buf, "TracerPid:\t0\n");
            }
        }
        return ret;
    }, "pointer", ["pointer", "int", "pointer"]));
});

```
