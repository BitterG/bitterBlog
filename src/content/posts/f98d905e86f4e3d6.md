## hook 动态加载 dex
```javascript
Java.perform(function(){
      Java.enumerateClassLoaders({
      onMatch : function(loader){
            try {
                  // loadClass or findClass
                  if (loader.loadClass("com.example.myapplication.MyTest")){
                  Java.classFactory.loader = loader;	//切换加载器上下文
                  var hookClass = Java.use("com.example.myapplication.MyTest");
                  console.log("success hook it :", hookClass);
                  // something to do;
                  }
            } catch (error) {
                  // pass
            }
      },
      
      onComplete: function () {
            console.log("complete !!! ")
      }
      })
})
```

解释

:::info
1. Java.perform(function(){ ... })  保证所有 Java API 可用后再执行脚本。

2. Java.enumerateClassLoaders({ ... })  遍历所有已加载的 ClassLoader。

3. onMatch: function(loader){ ... }  对每个 ClassLoader 执行回调，尝试用它加载目标类 

4. if (loader.loadClass("..."))  如果当前 loader 能加载目标类，说明这个 loader 就是目标类的 ClassLoader。

5. Java.classFactory.loader = loader;  切换 Frida 的 Java.use/Java.choose 等操作到这个 ClassLoader，确保后续 hook 的是正确的类。

6. var hookClass = Java.use("xxx.xxx.XX");  获取目标类的 JavaScript 代理对象，可以进行 hook 或调用。

7. console.log("success hook it :", hookClass);  打印 hook 成功信息。

8. onComplete: function () { ... }  所有 ClassLoader 遍历完成后执行，打印 "complete !!!"。



Java.classFactory.loader = loader;

这句的作用是切换 Frida 的类加载器（ClassLoader）上下文，让你后续用 Java.use、Java.choose 等操作时，能够访问到指定 ClassLoader 加载的类。



 为什么要这样写？

在 Android 应用中，可能存在多个 ClassLoader（如主程序、插件、热修复、WebView等），有些类是动态加载的，不在默认的 ClassLoader 里。如果你直接用 Java.use("xxx")，Frida 只会在默认 ClassLoader 里找，找不到就会报错。



切换 loader 后：



你可以 hook、调用那些只在特定 ClassLoader 里的类。

解决“找不到类”或“类未定义”的问题。

举例说明

假如 com.xxx.xx.Xxxx 这个类是插件动态加载的，只有插件的 ClassLoader 能访问。你用默认 loader 是找不到的。

这时你遍历所有 loader，找到能加载这个类的 loader。

:::

```javascript
Java.perform(function () {
      Java.enumerateClassLoadersSync().forEach(function (classloader) {
            try {
                  console.log("classloader", classloader);
                  classloader.loadClass("com.example.myapplication.MyTest");
                  Java.classFactory.loader = classloader;
                  var mainActivityClass = Java.use("com.example.myapplication.MyTest");
                  console.log("mainActivityClass", mainActivityClass);
            } catch (error) {
                  console.log("error", error);
            }
      });
})
```

## <font style="color:rgb(31, 45, 61);">Hook 监控控件 onClick</font>
```javascript
var jclazz = null;
var jobj = null;

function getObjClassName(obj) {
    if (!jclazz) {
        var jclazz = Java.use("java.lang.Class");
    }
    if (!jobj) {
        var jobj = Java.use("java.lang.Object");
    }
    return jclazz.getName.call(jobj.getClass.call(obj));
}

function watch(obj, mtdName) {
    var listener_name = getObjClassName(obj);
    var target = Java.use(listener_name);
    if (!target || !mtdName in target) {
        return;
    }
    // send("[WatchEvent] hooking " + mtdName + ": " + listener_name);
    target[mtdName].overloads.forEach(function (overload) {
        overload.implementation = function () {
            //send("[WatchEvent] " + mtdName + ": " + getObjClassName(this));
            console.log("[WatchEvent] " + mtdName + ": " + getObjClassName(this))
            return this[mtdName].apply(this, arguments);
        };
    })
}

function OnClickListener() {
    Java.perform(function () {

        //以spawn启动进程的模式来attach的话
        Java.use("android.view.View").setOnClickListener.implementation = function (listener) {
            if (listener != null) {
                watch(listener, 'onClick');
            }
            return this.setOnClickListener(listener);
        };

        //如果frida以attach的模式进行attch的话
        Java.choose("android.view.View$ListenerInfo", {
            onMatch: function (instance) {
                instance = instance.mOnClickListener.value;
                if (instance) {
                    console.log("mOnClickListener name is :" + getObjClassName(instance));
                    watch(instance, 'onClick');
                }
            },
            onComplete: function () {
            }
        })
    })
}
```

```javascript
    /**
     * Base64编码 - 字符串
     * @param input 需要编码的字符串
     * @return Base64编码后的字符串
     */
    public static String encode(String input) {
        if (input == null) {
            return null;
        }
        return Base64.getEncoder().encodeToString(input.getBytes(StandardCharsets.UTF_8));
    }
    
    /**
     * Base64编码 - 字节数组
     * @param input 需要编码的字节数组
     * @return Base64编码后的字符串
     */
    public static String encode(byte[] input) {
        if (input == null) {
            return null;
        }
        return Base64.getEncoder().encodeToString(input);
    }
    
    /**
     * Base64解码 - 返回字符串
     * @param encodedInput Base64编码的字符串
     * @return 解码后的原始字符串
     */
    public static String decode(String encodedInput) {
        if (encodedInput == null) {
            return null;
        }
        try {
            byte[] decodedBytes = Base64.getDecoder().decode(encodedInput);
            return new String(decodedBytes, StandardCharsets.UTF_8);
        } catch (IllegalArgumentException e) {
            throw new RuntimeException("Invalid Base64 input", e);
        }
    }
    
    /**
     * Base64解码 - 返回字节数组
     * @param encodedInput Base64编码的字符串
     * @return 解码后的字节数组
     */
    public static byte[] decodeToBytes(String encodedInput) {
        if (encodedInput == null) {
            return null;
        }
        try {
            return Base64.getDecoder().decode(encodedInput);
        } catch (IllegalArgumentException e) {
            throw new RuntimeException("Invalid Base64 input", e);
        }
    }

     * URL安全的Base64编码
     * @param input 需要编码的字符串

    public static String encodeUrlSafe(String input) {

            return null;
        }
```
