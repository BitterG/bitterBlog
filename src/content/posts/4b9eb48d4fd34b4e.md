# native hook

以修改app中so里的指定函数的返回值为例



1.在android studio中为xposed模块项目mian目录右键添加"Add C++ to Module"，之后即会在main目录下创建一个cpp文件夹(之后build.gradle中会自动添加一些配置, 有需要可以进行更改, 一般默认即可)

2.在AndroidManifest.xml下额外添加两行

```xml
<application ...
  android:multiArch="true"
  android:extractNativeLibs="false"
  >
```

3.正常创建一个xposed模块, 模块中加载之后要注入的so

```java
public class HookMain implements IXposedHookLoadPackage {

    private static final String TAG = "MyLSP";
    
    @Override
    public void handleLoadPackage(XC_LoadPackage.LoadPackageParam lpparam) throws Throwable {
        //判断是否为需要注入的目标应用
        if ("com.example.myapplication".equals(lpparam.packageName)) {
            try {
                //加载准备注入的so
                Log.i(TAG, "加载my so文件");
                //记载自己的编写测so注入文件 libxxxxxx.so
                System.loadLibrary("mylsposedmod");
            } catch (Exception e) {
                Log.i(TAG, "error " + e.getMessage());
            }
        }
    }
}

```

4.创建编译需要的.h头文件，名称自定

```cpp
typedef int (*HookFunType)(void *func, void *replace, void **backup);

typedef int (*UnhookFunType)(void *func);

typedef void (*NativeOnModuleLoaded)(const char *name, void *handle);

typedef struct {
uint32_t version;
HookFunType hook_func;
UnhookFunType unhook_func;
} NativeAPIEntries;

typedef NativeOnModuleLoaded (*NativeInit)(const NativeAPIEntries *entries);
```

6.编写hook代码！

```cpp
#include <jni.h>
#include <string>
#include <dlfcn.h>
#include "head.h"
static HookFunType hook_func = nullptr;

//根据实际需要hook的函数编写
double (*back_calculateSquareRoot)(double a1);
//替代函数
double fake_calculateSquareRoot(double a1) {
    // 强制返回固定值，或者修改原始逻辑
    double result = back_calculateSquareRoot(a1);
    //修改最后返回值
    return result + 100000;
}

FILE *(*backup_fopen)(const char *filename, const char *mode);

FILE *fake_fopen(const char *filename, const char *mode) {
    if (strstr(filename, "banned")) return nullptr;
    return backup_fopen(filename, mode);
}

jclass (*backup_FindClass)(JNIEnv *env, const char *name);
jclass fake_FindClass(JNIEnv *env, const char *name)
{
    if(!strcmp(name, "dalvik/system/BaseDexClassLoader"))
        return nullptr;
    return backup_FindClass(env, name);
}
static inline bool ends_with(const std::string& str, const std::string& suffix) {
    return str.size() >= suffix.size() &&
           str.compare(str.size() - suffix.size(), suffix.size(), suffix) == 0;
}

void on_library_loaded(const char *name, void *handle) {
    // 指定app中的目标so
    if (ends_with(name, "libnative-lib.so")) {
        //指定方法
        void *target = dlsym(handle, "Java_com_example_myapplication_Java2CppTest_calculateSquareRoot");
        //                         自实现替代函数             函数指针声明
        hook_func(target, (void *) fake_calculateSquareRoot, (void **) &back_calculateSquareRoot);
    }
}

extern "C" [[gnu::visibility("default")]] [[gnu::used]]
jint JNI_OnLoad(JavaVM *jvm, void*) {
    JNIEnv *env = nullptr;
    jvm->GetEnv((void **)&env, JNI_VERSION_1_6);
    hook_func((void *)env->functions->FindClass, (void *)fake_FindClass, (void **)&backup_FindClass);
    return JNI_VERSION_1_6;
}

extern "C" [[gnu::visibility("default")]] [[gnu::used]]
NativeOnModuleLoaded native_init(const NativeAPIEntries *entries) {
    hook_func = entries->hook_func;
    // system hooks
    hook_func((void*) fopen, (void*) fake_fopen, (void**) &backup_fopen);
    return on_library_loaded;
}

```

app对应so的反汇编(IDA)

```cpp
double __fastcall Java_com_example_myapplication_Java2CppTest_calculateSquareRoot(double a1)
{
  double v1; // d8

  if ( a1 >= 0.0 )
  {
    v1 = sqrt(a1);
    __android_log_print(4LL, "NativeLib", "Square root of %f = %f");
  }
  else
  {
    __android_log_print(6LL, "NativeLib", "Cannot calculate square root of negative number: %f");
    v1 = -1.0;
  }
  return v1;
}
```

app源码主要部分

```cpp
public class Java2CppTest {

    private static final String TAG = "Java2CppTest";

    static  {
        //加载cpp库
        try {
            System.loadLibrary("native-lib");
            Log.d(TAG, "Native library loaded successfully");
        } catch (UnsatisfiedLinkError e) {
            Log.e(TAG, "Failed to load native library", e);
        }
    }

    // 声明native方法
    public native double calculateSquareRoot(double number);
}

```
