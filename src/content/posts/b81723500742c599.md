```c
// SPDX-License-Identifier: BSD-3-Clause
#ifndef BAD_BPF_COMMON_UM_H
#define BAD_BPF_COMMON_UM_H

#include <bpf/bpf.h>
#include <bpf/libbpf.h>
#include <unistd.h>
#include <signal.h>
#include <sys/resource.h>
#include <errno.h>
#include <fcntl.h>

static volatile sig_atomic_t exiting;

void sig_int(int signo)
{
    exiting = 1;
}

static bool setup_sig_handler() {
    // Add handlers for SIGINT and SIGTERM so we shutdown cleanly
    __sighandler_t sighandler = signal(SIGINT, sig_int);
    if (sighandler == SIG_ERR) {
        fprintf(stderr, "can't set signal handler: %s\n", strerror(errno));
        return false;
    }
    sighandler = signal(SIGTERM, sig_int);
    if (sighandler == SIG_ERR) {
        fprintf(stderr, "can't set signal handler: %s\n", strerror(errno));
        return false;
    }
    return true;
}

static int libbpf_print_fn(enum libbpf_print_level level, const char *format, va_list args)
{
    return vfprintf(stderr, format, args);
}

static bool bump_memlock_rlimit(void)
{
    struct rlimit rlim_new = {
        .rlim_cur    = RLIM_INFINITY,
        .rlim_max    = RLIM_INFINITY,
    };

    if (setrlimit(RLIMIT_MEMLOCK, &rlim_new)) {
        fprintf(stderr, "Failed to increase RLIMIT_MEMLOCK limit! (hint: run as root)\n");
        return false;
    }
    return true;
}


static bool setup() {
    // Set up libbpf errors and debug info callback 
    libbpf_set_print(libbpf_print_fn);

    // Bump RLIMIT_MEMLOCK to allow BPF sub-system to do anything 
    if (!bump_memlock_rlimit()) {
        return false;
    };

    // Setup signal handler so we exit cleanly
    if (!setup_sig_handler()) {
        return false;
    }

    return true;
}


#ifdef BAD_BPF_USE_TRACE_PIPE
static void read_trace_pipe(void) {
    int trace_fd;

    trace_fd = open("/sys/kernel/debug/tracing/trace_pipe", O_RDONLY, 0);
    if (trace_fd == -1) {
        printf("Error opening trace_pipe: %s\n", strerror(errno));
        return;
    }

    while (!exiting) {
        static char buf[4096];
        ssize_t sz;

        sz = read(trace_fd, buf, sizeof(buf) -1);
        if (sz > 0) {
            buf[sz] = '\x00';
            puts(buf);
        }
    }
}
#endif  // BAD_BPF_USE_TRACE_PIPE

#endif  // BAD_BPF_COMMON_UM_H
```

用户态代码 -p （可指定进程id生效） -r /a/b/c:/x/y/z (文件1:文件2 指定定向目标)

```c
// SPDX-License-Identifier: (LGPL-2.1 OR BSD-2-Clause)
/* Copyright (c) 2024 */
#include <argp.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <unistd.h>
#include <sys/resource.h>
#include <bpf/libbpf.h>
#include "common.h"
#include "common_um.h"
#include "file_redirection.skel.h"

#define EVENT_FILE_REDIRECT 1
#define EVENT_FILE_READ 2
#define EVENT_FILE_CLOSE 3
#define EVENT_PID_FILTERED 4

static struct env {
    char **redirect_pairs;
    int num_pairs;
    int *target_pids;
    int num_pids;
} env;

const char argp_program_doc[] =
"File Redirection\n"
"\n"
"Uses eBPF to redirect file operations from one path to another\n"
"\n"
"USAGE: ./file_redirection -r /path/original:/path/redirected [-r /another/path:/redirect/path] \n";

static const struct argp_option opts[] = {
    { "redirect", 'r', "ORIG:REDIR", 0, "File redirection pair (original:redirected). Can be specified multiple times.", 0 },
    { "pid", 'p', "PID", 0, "Target PID to affect. Can be specified multiple times. ", 0 },
    {},
};

static error_t parse_arg(int key, char *arg, struct argp_state *state)
{
    switch (key) {
    case 'r': {
        env.num_pairs++;
        env.redirect_pairs = realloc(env.redirect_pairs, env.num_pairs * sizeof(char *));
        if (!env.redirect_pairs) {
            fprintf(stderr, "Failed to allocate memory for redirect pairs\n");
            exit(1);
        }

        env.redirect_pairs[env.num_pairs - 1] = strdup(arg);
        if (!env.redirect_pairs[env.num_pairs - 1]) {
            fprintf(stderr, "Failed to allocate memory for redirect pair\n");
            exit(1);
        }
        break;
    }
    case 'p': {
        errno = 0;
        int pid = strtol(arg, NULL, 10);
        if (errno || pid <= 0) {
            fprintf(stderr, "Invalid pid: %s\n", arg);
            argp_usage(state);
        }

        env.num_pids++;
        env.target_pids = realloc(env.target_pids, env.num_pids * sizeof(int));
        if (!env.target_pids) {
            fprintf(stderr, "Failed to allocate memory for target PIDs\n");
            exit(1);
        }

        env.target_pids[env.num_pids - 1] = pid;
        break;
    }
    case ARGP_KEY_ARG:
        argp_usage(state);
        break;
    default:
        return ARGP_ERR_UNKNOWN;
    }
    return 0;
}

static const struct argp argp = {
    .options = opts,
    .parser = parse_arg,
    .doc = argp_program_doc,
};

static int handle_event(void *ctx, void *data, size_t data_sz)
{
    const struct event *e = data;

    if (env.num_pids > 0) {
        bool pid_found = false;
        for (int i = 0; i < env.num_pids; i++) {
            if (e->pid == env.target_pids[i]) {
                pid_found = true;
                break;
            }
        }

        if (!pid_found) {
            return 0;
        }
    }

    if (e->sucess) {
        switch (e->event_type) {
        case EVENT_FILE_REDIRECT:
            printf("Redirected file access for process %d (%s)\n", e->pid, e->comm);
            printf("  Original path: %s\n", e->orig_filename);
            printf("  Redirected to: %s\n", e->redirect_filename);
            break;
        case EVENT_FILE_READ:
            printf("File read by process %d (%s)\n", e->pid, e->comm);
            printf("  File path: %s\n", e->orig_filename);
            printf("  File descriptor: %d\n", e->fd);
            break;
        default:
            break;
        }
    } else {
        switch (e->event_type) {
        case EVENT_PID_FILTERED:
            printf("PID filtered: %d (%s) - not in target list\n", e->pid, e->comm);
            break;
        default:
            printf("Failed operation for process %d (%s)\n", e->pid, e->comm);
            break;
        }
    }
    return 0;
}

static int parse_redirect_pair(const char *pair, struct tr_file *orig, struct tr_file *redir)
{
    char *colon = strchr(pair, ':');
    if (!colon) {
        fprintf(stderr, "Invalid redirection pair format: %s (should be original:redirected)\n", pair);
        return -1;
    }

    size_t orig_len = colon - pair;
    size_t redir_len = strlen(colon + 1);

    if (orig_len >= FILENAME_LEN_MAX || redir_len >= FILENAME_LEN_MAX) {
        fprintf(stderr, "Path too long in redirection pair: %s\n", pair);
        return -1;
    }

    memcpy(orig->filename, pair, orig_len);
    orig->filename[orig_len] = '\0';
    orig->filename_len = orig_len + 1;

    memcpy(redir->filename, colon + 1, redir_len);
    redir->filename[redir_len] = '\0';
    redir->filename_len = redir_len + 1;

    return 0;
}

int main(int argc, char **argv)
{
    struct ring_buffer *rb = NULL;
    struct file_redirection_bpf *skel;
    int err;

    memset(&env, 0, sizeof(env));

    err = argp_parse(&argp, argc, argv, 0, NULL, NULL);
    if (err)
        return err;

    if (env.num_pairs == 0) {
        fprintf(stderr, "At least one redirection pair (-r) must be specified\n");
        return 1;
    }

    if (!setup()) {
        fprintf(stderr, "Failed to do common setup\n");
        return 1;
    };

    skel = file_redirection_bpf__open();
    if (!skel) {
        fprintf(stderr, "Failed to open and load BPF skeleton\n");
        return 1;
    }

    if (env.num_pids > 0) {
        skel->rodata->use_pid_filter = 1;
    }

    err = file_redirection_bpf__load(skel);
    if (err) {
        fprintf(stderr, "Failed to load and verify BPF skeleton\n");
        goto cleanup;
    }

    if (env.num_pids > 0) {
        for (int i = 0; i < env.num_pids; i++) {
            int pid = env.target_pids[i];
            int value = 1;

            err = bpf_map_update_elem(bpf_map__fd(skel->maps.target_pids), &pid, &value, BPF_ANY);
            if (err) {
                fprintf(stderr, "Failed to update target_pids map: %s\n", strerror(errno));
                goto cleanup;
            }
        }
    }

    err = file_redirection_bpf__attach(skel);
    if (err) {
        fprintf(stderr, "Failed to attach BPF skeleton\n");
        goto cleanup;
    }

    for (int i = 0; i < env.num_pairs; i++) {
        struct tr_file orig = {}, redir = {};

        if (parse_redirect_pair(env.redirect_pairs[i], &orig, &redir) < 0) {
            err = -1;
            goto cleanup;
        }

        err = bpf_map__update_elem(skel->maps.redirect_map, &orig, sizeof(orig),
                                   &redir, sizeof(redir), BPF_ANY);
        if (err) {
            fprintf(stderr, "Failed to update redirection map: %s\n", strerror(-err));
            goto cleanup;
        }

        printf("Added redirection: %s -> %s\n", orig.filename, redir.filename);
    }

    rb = ring_buffer__new(bpf_map__fd(skel->maps.rb), handle_event, NULL, NULL);
    if (!rb) {
        err = -1;
        fprintf(stderr, "Failed to create ring buffer\n");
        goto cleanup;
    }

    printf("Successfully started!\n");
    printf("File redirection active with %d redirection rules\n", env.num_pairs);

    if (env.num_pids > 0) {
        printf("Only affecting specific PIDs: ");
        for (int i = 0; i < env.num_pids; i++) {
            printf("%d", env.target_pids[i]);
            if (i < env.num_pids - 1) {
                printf(", ");
            }
        }
        printf("\n");
    } else {
        printf("Affecting all processes\n");
    }

    while (!exiting) {
        err = ring_buffer__poll(rb, 100 /* timeout, ms */);
        if (err == -EINTR) {
            err = 0;
            break;
        }
        if (err < 0) {
            printf("Error polling ring buffer: %d\n", err);
            break;
        }
    }

cleanup:
    ring_buffer__free(rb);
    file_redirection_bpf__destroy(skel);

    for (int i = 0; i < env.num_pairs; i++) {
        free(env.redirect_pairs[i]);
    }
    free(env.redirect_pairs);

    free(env.target_pids);

    return err < 0 ? -err : 0;
}
```

```c
// SPDX-License-Identifier: GPL-2.0 OR BSD-3-Clause
/* Copyright (c) 2024 */
#include "vmlinux.h"
#include <bpf/bpf_core_read.h>
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>
#include "common.h"

// Event types for logging
#define EVENT_FILE_REDIRECT 1
#define EVENT_FILE_READ 2
#define EVENT_FILE_CLOSE 3
#define EVENT_PID_FILTERED 4

char LICENSE[] SEC("license") = "Dual BSD/GPL";

struct {
  __uint(type, BPF_MAP_TYPE_RINGBUF);
  __uint(max_entries, 256 * 1024);
} rb SEC(".maps");

struct {
  __uint(type, BPF_MAP_TYPE_HASH);
  __uint(max_entries, 256);
  __type(key, struct tr_file);
  __type(value, struct tr_file);
} redirect_map SEC(".maps");

struct {
  __uint(type, BPF_MAP_TYPE_HASH);
  __uint(max_entries, 1024);
  __type(key, size_t); // pid_tgid + fd
  __type(value, struct tr_file);
} fd_map SEC(".maps");

struct {
  __uint(type, BPF_MAP_TYPE_HASH);
  __uint(max_entries, 1024);
  __type(key, size_t); // pid_tgid + fd
  __type(value, struct tr_file);
} file_path_map SEC(".maps");

struct {
  __uint(type, BPF_MAP_TYPE_HASH);
  __uint(max_entries, 64);
  __type(key, int);
  __type(value, int);
} target_pids SEC(".maps");

const volatile int use_pid_filter = 0;

static inline size_t make_fd_key(size_t pid_tgid, u32 fd) {
  return (pid_tgid & 0xFFFFFFFF00000000) | (fd & 0xFFFFFFFF);
}

// #define __NR_openat 56
// #define __NR_newfstatat 79
// #define __NR_access 21
// #define __NR_close 57
#define __NR_openat 257      // 修正：x86_64 的 openat 是 257
#define __NR_newfstatat 262  // 修正：x86_64 的 newfstatat 是 262
#define __NR_access 21
#define __NR_close 3         // 修正：close 是 3

//重定向函数
static inline int handle_file_redirection(const char *pathname, u32 pid) {
  if (!pathname) {  //检查空指针
    return 0;
  }
  char path[FILENAME_LEN_MAX];
  long pathlen = bpf_probe_read_user_str(path, sizeof(path), pathname); //从用户态读取路径到path
  if (pathlen <= 0) { //读取成功检查
    return 0;
  }
  if (pathlen >= FILENAME_LEN_MAX) { //长度检查
    pathlen = FILENAME_LEN_MAX;
  }

  u32 len = (u32) pathlen;
  struct tr_file orig_file = {}; //保存原始文件路径
  bpf_probe_read_kernel(&orig_file.filename, len, path);
  orig_file.filename_len = len;

  //重定向判断条件
  struct tr_file *redirect =  bpf_map_lookup_elem(&redirect_map, &orig_file); //读取map 返回指针
  if (!redirect) {
    return 0;
  }

  char redirect_filename[FILENAME_LEN_MAX];
  bpf_probe_read_kernel(redirect_filename, FILENAME_LEN_MAX, (void *) redirect);

  //写回用户态
  bpf_probe_write_user((void *) pathname, redirect_filename, FILENAME_LEN_MAX);

  struct event *e;
  e = bpf_ringbuf_reserve(&rb, sizeof(*e), 0);
  if (e) {
    e->pid = pid;
    bpf_get_current_comm(&e->comm, sizeof(e->comm));
    e->sucess = true;
    e->event_type = EVENT_FILE_REDIRECT;

    bpf_probe_read_kernel(e->orig_filename, len < FILENAME_LEN_MAX ? len : FILENAME_LEN_MAX - 1, path);
    e->orig_filename[FILENAME_LEN_MAX - 1] = '\0'; //防止c字符串超出长度出现问题

    u32 redirect_len = redirect->filename_len;
    if (redirect_len >= FILENAME_LEN_MAX) {
      redirect_len = FILENAME_LEN_MAX - 1;
    }

    bpf_probe_read_kernel(e->redirect_filename, redirect_len, redirect->filename);
    e->redirect_filename[FILENAME_LEN_MAX - 1] = '\0';

    bpf_ringbuf_submit(e, 0);
  }

  return 1;
}

// Hook syscall entry
SEC("tracepoint/raw_syscalls/sys_enter")
int handle_syscall_enter(struct trace_event_raw_sys_enter *ctx) {
  size_t pid_tgid = bpf_get_current_pid_tgid();
  u32 pid = pid_tgid >> 32;
  u32 syscall_id = ctx->id;

  if (use_pid_filter) {
    int pid_key = pid;
    int *found = bpf_map_lookup_elem(&target_pids, &pid_key);

    if (!found) {
      return 0;
    }
  }
  // todo: impl 
  if (syscall_id == __NR_openat) {
    const char *pathname = (const char *) ctx->args[1];
    char path[FILENAME_LEN_MAX];
    long pathlen = bpf_probe_read_user_str(path, sizeof(path), pathname);
    if (pathlen <= 0) { //读取成功检查
      return 0;
    }
    if (pathlen >= FILENAME_LEN_MAX) { //长度检查
      pathlen = FILENAME_LEN_MAX;
    }

    //存入file_info结构体
    u32 len = (u32) pathlen;
    size_t key = pid_tgid;
    struct tr_file file_info = {};
    bpf_probe_read_kernel(&file_info.filename, len, path);
    file_info.filename_len = len;

    bpf_map_update_elem(&file_path_map, &key, &file_info, BPF_ANY);
    return handle_file_redirection(pathname, pid);
  } else if (syscall_id == __NR_newfstatat) {
    int dirfd = (int) ctx->args[0];
    const char *pathname = (const char *) ctx->args[1];
    return handle_file_redirection(pathname, pid);
  } else if (syscall_id == __NR_access) {
    const char *pathname = (const char *) ctx->args[0];
    return handle_file_redirection(pathname, pid);
  }

  return 0;
}

// Hook syscall exit to track file descriptors
SEC("tracepoint/raw_syscalls/sys_exit")
int handle_syscall_exit(struct trace_event_raw_sys_exit *ctx) {
  u32 syscall_id = ctx->id;
  s32 ret = ctx->ret;

  // Only handle openat syscall exit
  if (syscall_id != __NR_openat)
    return 0;

  // Only track successful opens with valid fds
  if (ret < 0)
    return 0;

  // todo: impl
  size_t pid_tgid = bpf_get_current_pid_tgid();
  u32 pid = pid_tgid >> 32;
  size_t path_key = pid_tgid;
  struct tr_file *file_info = bpf_map_lookup_elem(&file_path_map, &path_key);

  struct event *e;
  e = bpf_ringbuf_reserve(&rb, sizeof(*e), 0);
  if (e) {
    e->pid = pid;
    bpf_get_current_comm(&e->comm, sizeof(e->comm));
    e->sucess = true;
    e->event_type = EVENT_FILE_READ;
    e->fd = ret;

    if (file_info) {
      u32 len = file_info->filename_len;
      bpf_probe_read_kernel(e->orig_filename, FILENAME_LEN_MAX, file_info->filename);
      e->orig_filename[FILENAME_LEN_MAX - 1] = '\0';
      size_t fd_key = make_fd_key(pid_tgid, (u32)ret);
      bpf_map_update_elem(&fd_map, &fd_key, file_info, BPF_ANY);

      bpf_map_delete_elem(&file_path_map, &path_key);
    }

    bpf_ringbuf_submit(e, 0);
  }
  return 0;
}

// Hook close syscall to clean up fd tracking
SEC("tracepoint/raw_syscalls/sys_enter")
int handle_close_enter(struct trace_event_raw_sys_enter *ctx) {
  u32 syscall_id = ctx->id;

  // Only handle close syscall
  if (syscall_id != __NR_close)
    return 0;

  // todo: impl
  size_t pid_tgid = bpf_get_current_pid_tgid();
  u32 pid = pid_tgid >> 32;
  u32 fd = (u32) ctx->args[0];
  size_t fd_key = make_fd_key(pid_tgid, fd);
  struct tr_file *file_info = bpf_map_lookup_elem(&fd_map, &fd_key);
  if (file_info) {
    struct event *e;
    e = bpf_ringbuf_reserve(&rb, sizeof(*e), 0);
    if (e) {
      e->pid = pid;
      bpf_get_current_comm(&e->comm, sizeof(e->comm));
      e->sucess = true;
      e->event_type = EVENT_FILE_CLOSE;
      e->fd = fd;

      u32 len = file_info->filename_len;
      bpf_probe_read_kernel(e->orig_filename, FILENAME_LEN_MAX, file_info->filename);
      e->orig_filename[FILENAME_LEN_MAX - 1] = '\0';

      bpf_ringbuf_submit(e, 0);
    }

    bpf_map_delete_elem(&fd_map, &fd_key);
  }
  return 0;
}
```
