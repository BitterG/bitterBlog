# java操作Zh

1.导入依赖

```java
<dependency>
    <groupId>com.101tec</groupId>
    <artifactId>zkclient</artifactId>
    <version>0.10</version>
</dependency>
```

2.创建客户端对象

```java
private static ZkClient zkClient = new ZkClient("地址:端口", 60000 * 30, 60000, new SerializableSerializer());
```

3.例子（之前学习Zh时写的测试）

```java
package com.bitter.zookpeeperDemo;

import org.I0Itec.zkclient.IZkDataListener;
import org.I0Itec.zkclient.ZkClient;
import org.I0Itec.zkclient.serialize.SerializableSerializer;
import org.apache.zookeeper.CreateMode;
import org.apache.zookeeper.data.Stat;
import org.junit.jupiter.api.Test;

import java.io.IOException;
import java.util.List;

public class ZkTest {

    private static ZkClient zkClient;
    static {
        zkClient = new ZkClient("192.168.200.129:2181", 60000 * 30, 60000, new SerializableSerializer());
    }

    @Test
    public void  test01(){
        zkClient = new ZkClient("192.168.200.129:2181", 60000 * 30, 60000, new SerializableSerializer());
        System.out.println(zkClient);
        zkClient.close();
    }

    @Test
    public void testCreateNode() {
        //创建持久节点
        zkClient.create("/app2", "xue", CreateMode.PERSISTENT);
        //创建持久顺序节点
        zkClient.create("/app3","yue", CreateMode.PERSISTENT_SEQUENTIAL);
        //创建临时节点
        zkClient.create("/app4","qing", CreateMode.EPHEMERAL);
        //创建临时顺序节点
        zkClient.create("/app5","haha",CreateMode.EPHEMERAL_SEQUENTIAL);
    }

    @Test
    public void test2(){
        //查看某个节点数据，需要保证创建和获取的数据序列化方式相同，如果是在linux上创建的节点那么在Java上获取就会出现序列化不匹配的异常

        //查看节点状态
        Stat stat = new Stat();
        Object o = zkClient.readData("/app2", stat);

        System.out.println(o);
        //查看节点创建时间
        System.out.println(stat.getCtime());
        //查看版本
        System.out.println(stat.getCversion());
        //查看id
        System.out.println(stat.getCzxid());
    }

    @Test
    public void test3() {
        zkClient.writeData("/app2", "newData Hello Zookeeper!");
    }
    @Test
    public void test6() {
        zkClient.delete("/app2");
    }

    @Test
    public void test4(){

        List<String> children = zkClient.getChildren("/");
        for(String c : children){
            System.out.println(c);
        }

    }

    @Test
    public  void test5() throws IOException {

        zkClient.subscribeDataChanges("/app2", new IZkDataListener() {
            //当前节点数据变化时触发
            @Override
            public void handleDataChange(String dataPath, Object o) throws Exception {
                System.out.println("监听到节点数发生变化");
                System.out.println("当前节点路径"+dataPath);
                System.out.println("当前节点信息"+o);
            }
            //当前节点删除时触发
            @Override
            public void handleDataDeleted(String dataPath) throws Exception {
                System.out.println("监听到节点被杀删除");
                System.out.println("当前节点路径"+dataPath);
            }
        });
        //阻塞当前监听
        System.in.read();
    }

    //递归创建父节点
    @Test
    public void createNodeTest() {
        zkClient.createPersistent("/a/b/c/d/e/f", true);

    }

    @Test
    public void  getChileTest() {
        getAllChile("/a");

    }

    //递归获取所有子节点
    public void getAllChile(String path) {
        System.out.println(path);
        List<String> childrenList = zkClient.getChildren(path);
        if (childrenList == null || childrenList.isEmpty()) return;
        for (String s : childrenList) {
            if (path.equals("/")) {
                getAllChile(path + s);
            } else {
                getAllChile(path + "/" + s);
            }
        }
    }


}
```
