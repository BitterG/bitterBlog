```javascript
# -*- coding: utf-8 -*-
import requests
import time
import execjs

with open('md5_mod.js', 'r', encoding='utf-8') as f:
    js_code = f.read()
ctx = execjs.compile(js_code)

par_t = int(time.time() * 1000)
par_sign = ctx.call("sign_hex", "1:" + str(par_t))

url = "https://www.python-spider.com/api/app10"

sum = 0
fail_list = []


for i in range(1, 101):   

      par_t = int(time.time() * 1000)
      par_sign = ctx.call("sign_hex", str(i) + ":" + str(par_t))
      
      payload = {
            "page": i,
            "t": par_t,
            "sign": par_sign
      }

      response = requests.post(url, json=payload)
      # print("input: " + str(payload['page']) + ":" + str(payload['t']), " output: " + payload['sign'])
      print(response.json())
      data = response.json().get('data', [])
      if not data:
            print("No data found for page", i)
            fail_list.append(i)
            continue
      
      for v in data:
           sum += int(v['value'])

#重试
while len((fail_list)) > 0:
 
      for i in fail_list:
            par_t = int(time.time() * 1000)
            par_sign = ctx.call("sign_hex", str(i) + ":" + str(par_t))
            
            payload = {
                  "page": i,
                  "t": par_t,
                  "sign": par_sign
            }

            response = requests.post(url, json=payload)
            data = response.json().get('data', [])
            if not data:
                  print("No data found for page", i)
                  continue
            fail_list.remove(i)
            for v in data:
                  sum += int(v['value'])

print("总和:", sum)
```
