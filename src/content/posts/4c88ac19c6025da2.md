# SpringBoot整合mongoDB

## 添加依赖与配置文件
1.导入maven坐标

```plain
<dependency>
		<groupId>org.springframework.boot</groupId>
  	<artifactId>spring-boot-starter-data-mongodb</artifactId>
    <version>2.3.9.RELEASE</version>
</dependency>

```

2.添加yml配置文件

```plain
spring:
  data:
    mongodb:
      uri: mongodb://用户名:密码@地址:端口号/数据库名?authSource=admin&authMechanism=SCRAM-SHA-1

      # 上方为明确指定某个数据的用户进行连接
      # 也可以使用admin 数据库中的用户进行连接  统一到admin 数据库进行认证
      # admin 用户认证 url 写法： mongodb://账户:密码%40@ip:端口/数据库名?authSource=admin&authMechanism=SCRAM-SHA-1
```

3.编写实体类

```java
package com.bitter.mangotest.entity;

import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;
import org.springframework.data.annotation.Id;
import org.springframework.data.mongodb.core.mapping.Document;

@Data
@NoArgsConstructor
@AllArgsConstructor
//Document注解对应到mongo数据库的集合名
@Document("User")
public class User {
    //使用@Id注解是id属性和数据库主键构成映射关系
    @Id
    private String id;
    //@Field(“person_name”)： //指定属性名和数据库域的映射关系 默认属性名
    private String name;
    private Integer age;
    private String email;
    private String createDate;

}
```

## 使用MongoTemplate
编写测试类

```java
@SpringBootTest
class MangoTestApplicationTests {

    //注入
    @Autowired
    private MongoTemplate mongoTemplate;

	//加入数据
    @Test
    void createData() {
        User user = new User();
        user.setName("张三");
        user.setAge(20);
        user.setEmail("123456@gmail.com");
        user.setCreateDate("太阳系地球中国");

        User user1 = mongoTemplate.insert(user);
        System.out.println(user1);
    }

    //查询所有数据
    @Test
    void findAll() {
        List<User> all = mongoTemplate.findAll(User.class);
        System.out.println(all);
    }

    //根据id查询
    @Test
    void findById() {
        User user = mongoTemplate.findById("63fca2ba634f7f75d14e0d9a", User.class);
        System.out.println(user);
    }

    //条件查询
    @Test
    void findUserList() {
        Query query = new Query(
                Criteria.where("name")
                        .is("张三")
                        .and("age").is(20)
        );
        List<User> users = mongoTemplate.find(query, User.class);
        System.out.println(users);
    }

    //模糊查询
    @Test
    void findLikeUserList() {

        //创建正则
        String name = "三";
        String regex = String.format("%s%s%s", "^.*", name, ".*$");
        Pattern pattern = Pattern.compile(regex, Pattern.CASE_INSENSITIVE);

        Query query = new Query(
                //正则匹配
                Criteria.where("name").regex(pattern)
        );

        List<User> users = mongoTemplate.find(query, User.class);
        System.out.println(users);
    }

    //分页查询
    @Test
    void findLikeUserPage() {
        //分页条件
        int pageNo = 1;
        int pageSize = 10;

        //创建正则
        String name = "三";
        String regex = String.format("%s%s%s", "^.*", name, ".*$");
        Pattern pattern = Pattern.compile(regex, Pattern.CASE_INSENSITIVE);

        Query query = new Query(
                //正则匹配
                Criteria.where("name").regex(pattern)
        );

        //根据条件查询总记录数
        long count = mongoTemplate.count(query, User.class);
        //分页查询
        List<User> users = mongoTemplate.find(
                query
                        .skip((pageNo - 1) * pageSize)
                        .limit(pageSize)
                , User.class);
        System.out.println(users);
    }

    //更新数据
    @Test
    void updateUser() {
        //根据id查询
        User user = mongoTemplate.findById("63fcb7c3e02faa385a3f2ebd", User.class);
        //修改值
        user.setName("李四");
        user.setAge(30);
        user.setEmail("9999@qq.com");
        //调用方法实现修改
        Query query = new Query(
                Criteria.where("_id").is(user.getId())
        );
        Update update = new Update();
        update.set("name", user.getName());
        update.set("age", user.getAge());
        update.set("email", user.getEmail());
        UpdateResult result = mongoTemplate.upsert(query, update, User.class);
        System.out.println(result);
    }

    //删除操作
    @Test
    void deleteUser() {
        Query query = new Query(Criteria.where("_id").is("63fcb7c3e02faa385a3f2ebd"));
        DeleteResult result = mongoTemplate.remove(query, User.class);
        long deletedCount = result.getDeletedCount();
        System.out.println(result);
        System.out.println(deletedCount);
    }

}
```

## 使用MongoRepository
1.继承MongoRepository接口

```java
package com.bitter.mangotest.repository;

import com.bitter.mangotest.entity.User;
import org.springframework.data.mongodb.repository.MongoRepository;

public interface UserRepository extends MongoRepository<User, String> {
}
```

2.编写测试方法

```java
package com.bitter.mangotest;

import com.bitter.mangotest.entity.User;
import com.bitter.mangotest.repository.UserRepository;

import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.data.domain.*;
import org.springframework.util.StringUtils;

import java.util.List;
import java.util.Optional;


@SpringBootTest
    public class TestMongoRepository {

        @Autowired
        private UserRepository userRepository;

        //    加入数据
        @Test
        void createData() {
            User user = new User();
            user.setName("老七");
            user.setAge(35);
            user.setEmail("6348@qq.com");
            user.setCreateDate("月球");
            User result = userRepository.save(user);

            System.out.println(result);
        }

        //查询所有数据
        @Test
        void findAll() {
            List<User> all = userRepository.findAll();
            System.out.println(all);
        }

        //根据id查询
        @Test
        void findById() {
            User temp = new User();
            String id = "63fd5043c9cb944c15d53195&";
            User user = userRepository.findById(id).orElse(temp);
            /*
            *  其返回的石Optional类型，该类型是为了规避NULL错误。
            *  应该使用orElse(“如果值为null的替代值”) 替代 get() 。
            */

            System.out.println(user);
        }

        //条件查询
        @Test
        void findUserList() {
            User user = new User();
            user.setName("老六");
            //构建条件
            Example<User> userExample = Example.of(user);

            //调用查询方法
            List<User> list = userRepository.findAll(userExample);
            System.out.println(list);
        }

        //模糊查询
        @Test
        void findLikeUserList() {
            User user = new User();
            user.setName("老");

            //设置模糊查询匹配规则
            ExampleMatcher matcher = ExampleMatcher.matching()
                .withStringMatcher(ExampleMatcher.StringMatcher.CONTAINING)
                .withIgnoreCase(true);
            //构建条件
            Example<User> userExample = Example.of(user, matcher);
            //调用查询方法
            //名称中包含“老”的User
            List<User> list = userRepository.findAll(userExample);
            System.out.println(list);
        }

        //分页查询
        @Test
        void findLikeUserPage() {
            //分页设置
            Pageable pageable = PageRequest.of(0, 3);

            User user = new User();
            user.setAge(35);

            //构建条件
            Example<User> userExample = Example.of(user);

            //调用查询方法
            Page<User> page = userRepository.findAll(userExample, pageable);

            System.out.println(page);
        }

        //更新操作
        @Test
        void updateUser() {
            User user = userRepository.findById("63fd5043c9cb944c15d53195").orElse(new User());
            if (StringUtils.hasText(user.getId())) {
                user.setName("老六2号");

                //save()方法有则修改无则添加一个新的
                User result = userRepository.save(user);
                System.out.println(result);
            }


        }

        //删除操作
        @Test
        void deleteUser() {
            userRepository.deleteById("63fd5043c9cb944c15d53195");
        }
}
```
